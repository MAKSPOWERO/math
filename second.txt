#include <iostream>
#include <fstream>
#include <vector>
#include <clocale>

using namespace std;

const int SIZE = 6;

// Функция для чтения матрицы из файла
bool readMatrixFromFile(const string& filename, vector<vector<int>>& matrix) {
    ifstream file(filename);
    if (!file.is_open()) {
        cout << "Ошибка открытия файла!" << endl;
        return false;
    }

    matrix.resize(SIZE, vector<int>(SIZE));
    for (int i = 0; i < SIZE; i++) {
        for (int j = 0; j < SIZE; j++) {
            file >> matrix[i][j];
        }
    }
    file.close();
    return true;
}

// Функция для проверки рефлексивности
bool isReflexive(const vector<vector<int>>& matrix) {
    for (int i = 0; i < SIZE; i++) {
        if (matrix[i][i] != 1) {
            return false;
        }
    }
    return true;
}

// Функция для проверки антирефлексивности
bool isIrreflexive(const vector<vector<int>>& matrix) {
    for (int i = 0; i < SIZE; i++) {
        if (matrix[i][i] != 0) {
            return false;
        }
    }
    return true;
}

// Функция для проверки симметричности
bool isSymmetric(const vector<vector<int>>& matrix) {
    for (int i = 0; i < SIZE; i++) {
        for (int j = 0; j < SIZE; j++) {
            if (matrix[i][j] != matrix[j][i]) {
                return false;
            }
        }
    }
    return true;
}

// Функция для проверки антисимметричности
bool isAntisymmetric(const vector<vector<int>>& matrix) {
    for (int i = 0; i < SIZE; i++) {
        for (int j = 0; j < SIZE; j++) {
            if (i != j && matrix[i][j] == 1 && matrix[j][i] == 1) {
                return false;
            }
        }
    }
    return true;
}

// Функция для проверки асимметричности
bool isAsymmetric(const vector<vector<int>>& matrix) {
    for (int i = 0; i < SIZE; i++) {
        for (int j = 0; j < SIZE; j++) {
            if (matrix[i][j] == 1 && matrix[j][i] == 1) {
                return false;
            }
        }
    }
    return true;
}

// Функция для проверки транзитивности
bool isTransitive(const vector<vector<int>>& matrix) {
    for (int i = 0; i < SIZE; i++) {
        for (int j = 0; j < SIZE; j++) {
            if (matrix[i][j] == 1) {
                for (int k = 0; k < SIZE; k++) {
                    if (matrix[j][k] == 1 && matrix[i][k] != 1) {
                        return false;
                    }
                }
            }
        }
    }
    return true;
}

// Функция для проверки связности
bool isConnected(const vector<vector<int>>& matrix) {
    for (int i = 0; i < SIZE; i++) {
        for (int j = 0; j < SIZE; j++) {
            if (i != j && matrix[i][j] == 0 && matrix[j][i] == 0) {
                return false;
            }
        }
    }
    return true;
}

// Функция для определения типа отношения эквивалентности
void checkEquivalence(const vector<vector<int>>& matrix) {
    if (isReflexive(matrix) && isSymmetric(matrix) && isTransitive(matrix)) {
        cout << "Отношение является ОТНОШЕНИЕМ ЭКВИВАЛЕНТНОСТИ" << endl;

        // Находим классы эквивалентности
        vector<bool> visited(SIZE, false);
        cout << "Классы эквивалентности:" << endl;

        for (int i = 0; i < SIZE; i++) {
            if (!visited[i]) {
                cout << "{ ";
                for (int j = 0; j < SIZE; j++) {
                    if (matrix[i][j] == 1) {
                        cout << j << " ";
                        visited[j] = true;
                    }
                }
                cout << "}" << endl;
            }
        }
    }
    else {
        cout << "Не является отношением эквивалентности" << endl;
    }
    cout << endl;
}

// Функция для определения типов отношений порядка
void checkOrderRelations(const vector<vector<int>>& matrix) {
    bool isPartialOrder = isReflexive(matrix) && isAntisymmetric(matrix) && isTransitive(matrix);
    bool isStrictPartialOrder = isIrreflexive(matrix) && isAsymmetric(matrix) && isTransitive(matrix);
    bool isLinearOrder = isPartialOrder && isConnected(matrix);
    bool isStrictLinearOrder = isStrictPartialOrder && isConnected(matrix);

    cout << "Типы отношений порядка:" << endl;
    cout << "Частичный порядок: " << (isPartialOrder ? "Да" : "Нет") << endl;
    cout << "Строгий частичный порядок: " << (isStrictPartialOrder ? "Да" : "Нет") << endl;
    cout << "Линейный порядок: " << (isLinearOrder ? "Да" : "Нет") << endl;
    cout << "Строгий линейный порядок: " << (isStrictLinearOrder ? "Да" : "Нет") << endl;
    cout << endl;
}

// Функция для вывода свойств отношения
void printProperties(const vector<vector<int>>& matrix) {
    cout << "Свойства отношения:" << endl;
    cout << "Рефлексивность: " << (isReflexive(matrix) ? "Да" : "Нет") << endl;
    cout << "Антирефлексивность: " << (isIrreflexive(matrix) ? "Да" : "Нет") << endl;
    cout << "Симметричность: " << (isSymmetric(matrix) ? "Да" : "Нет") << endl;
    cout << "Антисимметричность: " << (isAntisymmetric(matrix) ? "Да" : "Нет") << endl;
    cout << "Асимметричность: " << (isAsymmetric(matrix) ? "Да" : "Нет") << endl;
    cout << "Транзитивность: " << (isTransitive(matrix) ? "Да" : "Нет") << endl;
    cout << "Связность: " << (isConnected(matrix) ? "Да" : "Нет") << endl;
    cout << endl;
}

// Функция для вывода матрицы
void printMatrix(const vector<vector<int>>& matrix) {
    cout << "Матрица отношения:" << endl;
    for (int i = 0; i < SIZE; i++) {
        for (int j = 0; j < SIZE; j++) {
            cout << matrix[i][j] << " ";
        }
        cout << endl;
    }
    cout << endl;
}

int main() {
    setlocale(LC_ALL, "ru");
    vector<vector<int>> matrix;

    // Чтение матрицы из файла
    if (!readMatrixFromFile("matrix.txt", matrix)) {
        return 1;
    }

    // Вывод матрицы
    printMatrix(matrix);

    // Определение и вывод свойств
    printProperties(matrix);

    // Проверка на отношение эквивалентности
    checkEquivalence(matrix);

    // Проверка на отношения порядка
    checkOrderRelations(matrix);

    return 0;
}
